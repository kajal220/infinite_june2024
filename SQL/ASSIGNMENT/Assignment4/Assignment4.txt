/*--1.Write a T-SQL Program to find the factorial of a given number.*/

begin
declare @n int
set @n = 4
declare @fact bigint=1
declare @i int =1
while @i<=@n
begin
  set @fact=@fact*@i;
  set @i=@i+1;
end
print 'Factorial of ' + cast(@n as nvarchar(10)) + ' is ' + cast(@fact as nvarchar(20));
end
